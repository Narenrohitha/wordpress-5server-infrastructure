### `STEP 01` — Install & Secure MySQL
> 🎯 ![](https://img.shields.io/badge/Run_on-DB_SERVER-4479A1?style=flat-square&logo=mysql&logoColor=white)

```bash
sudo apt update && sudo apt upgrade -y
sudo apt install mysql-server -y
sudo systemctl start mysql && sudo systemctl enable mysql
sudo mysql_secure_installation
```

### `STEP 02` — Create Databases & Remote Users

```sql
sudo mysql -u root -p

CREATE DATABASE woocommerce_db  CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
CREATE DATABASE woocommerce_db3 CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE USER 'woo_user'@'[WEB1_IP]'  IDENTIFIED BY 'StrongPassword123!';
CREATE USER 'woo_user3'@'[WEB2_IP]' IDENTIFIED BY 'StrongPassword123!';

GRANT ALL PRIVILEGES ON woocommerce_db.*  TO 'woo_user'@'[WEB1_IP]';
GRANT ALL PRIVILEGES ON woocommerce_db3.* TO 'woo_user3'@'[WEB2_IP]';
FLUSH PRIVILEGES;
```

> ⚠️ ![](https://img.shields.io/badge/⚠️_#1_Mistake-DB__NAME_mismatch-red?style=flat-square) `DB_NAME` in `GRANT` must **exactly** match `wp-config.php` — or WordPress fails silently.

### `STEP 03` — Enable Remote Access + Slow Query Log

```ini
# /etc/mysql/mysql.conf.d/mysqld.cnf
bind-address        = 0.0.0.0
slow_query_log      = 1
slow_query_log_file = /var/log/mysql/mysql-slow.log
long_query_time     = 2
log_error           = /var/log/mysql/error.log
```

### `STEP 04` — Firewall — Allow Only Web Servers

```bash
sudo ufw allow OpenSSH
sudo ufw allow from [WEB1_IP] to any port 3306
sudo ufw allow from [WEB2_IP] to any port 3306
sudo ufw allow from [ELK_IP]  to any port 9100
sudo ufw enable
sudo systemctl restart mysql
```

> 💡 **Test remote connection BEFORE touching WordPress:**
> ```bash
> mysql -u woo_user -p -h [DB_IP] woocommerce_db -e "SELECT 1;"
> ```
> Any error here = firewall or GRANT issue. Fix it now — not after WordPress install.

### `STEP 08` — Round-Robin Config

```nginx
# /etc/nginx/conf.d/loadbalancer.conf
upstream wordpress_backend {
    server [WEB1_IP]:80;
    server [WEB2_IP]:80;
    keepalive 32;
}
server {
    listen 80;
    server_name [LB_IP];
    access_log /var/log/nginx/lb_access.log;
    error_log  /var/log/nginx/lb_error.log;
    location / {
        proxy_pass http://wordpress_backend;
        proxy_set_header Host             $host;
        proxy_set_header X-Real-IP        $remote_addr;
        proxy_set_header X-Forwarded-For  $proxy_add_x_forwarded_for;
        proxy_connect_timeout 60s;
        proxy_read_timeout    60s;
    }
}
```

### `STEP 09` — Domain-Based Routing + Prometheus stub_status

```nginx
# site1.com always → Web Server 1
server {
    listen 80;
    server_name site1.com www.site1.com;
    location / { proxy_pass http://[WEB1_IP]; }
}
# stub_status for Prometheus scraping
server {
    listen 127.0.0.1:8080;
    location /nginx_status {
        stub_status on;
        allow 127.0.0.1;
        deny all;
    }
}
```

```bash
sudo nginx -t && sudo systemctl reload nginx
```

### `STEP 12` — Logstash Pipeline

```ruby
# /etc/logstash/conf.d/main.conf
input { beats { port => 5044 } }

filter {
  if [log_type] == "nginx_lb" or [log_type] == "nginx_access" {
    grok  { match => { "message" => "%{COMBINEDAPACHELOG}" } }
    date  { match => ["timestamp","dd/MMM/yyyy:HH:mm:ss Z"] target => "@timestamp" }
    geoip { source => "clientip" }
  }
  if [log_type] == "mysql_error" {
    grok { match => { "message" => "%{TIMESTAMP_ISO8601} %{NUMBER} [%{DATA}] %{GREEDYDATA:msg}" } }
  }
}

output {
  elasticsearch {
    hosts    => ["http://localhost:9200"]
    user     => "elastic"
    password => "YourPassword"
    index    => "%{[log_type]}-%{+yyyy.MM.dd}"
  }
}
```

### `STEP 18` — Alert Rules

```yaml
# /etc/prometheus/alerts.yml
groups:
  - name: server_alerts
    rules:
      - alert: InstanceDown
        expr: up == 0
        for: 2m
        labels: { severity: critical }
        annotations:
          summary: "Instance {{ $labels.instance }} is DOWN"
      - alert: HighCPUUsage
        expr: 100-(avg by(instance)(rate(node_cpu_seconds_total{mode="idle"}[5m]))*100) > 80
        for: 5m
        labels: { severity: warning }
      - alert: HighMemoryUsage
        expr: (node_memory_MemTotal_bytes-node_memory_MemAvailable_bytes)/node_memory_MemTotal_bytes*100 > 85
        for: 5m
        labels: { severity: warning }
```

## 🩺 Troubleshooting

<img src="https://readme-typing-svg.demolab.com?font=JetBrains+Mono&size=13&duration=3000&pause=2000&color=F87171&center=true&vCenter=true&width=500&height=28&lines=Click+any+issue+to+expand+the+fix+👇"/>

</div>

<details>
<summary><b>🔴 &nbsp;"Error establishing a database connection"</b> &nbsp;<img src="https://img.shields.io/badge/Fix-DB__NAME_mismatch-red?style=flat-square"/></summary><br>

DB_NAME in `GRANT` must exactly match `wp-config.php`. `woocommerce_db ≠ woocommerce_db3`.
```bash
mysql -u woo_user -p -h [DB_IP] woocommerce_db -e "SELECT 1;"
# Fix any error here BEFORE touching WordPress
```
</details>

<details>
<summary><b>🟠 &nbsp;Elasticsearch not starting</b> &nbsp;<img src="https://img.shields.io/badge/Fix-vm.max__map__count-orange?style=flat-square"/></summary><br>

```bash
sudo sysctl -w vm.max_map_count=262144
echo "vm.max_map_count=262144" | sudo tee -a /etc/sysctl.conf
sudo systemctl restart elasticsearch
```
</details>

<details>
<summary><b>🟡 &nbsp;Filebeat not shipping logs</b> &nbsp;<img src="https://img.shields.io/badge/Fix-Test_output_first-yellow?style=flat-square"/></summary><br>

```bash
sudo filebeat test config
sudo filebeat test output
sudo journalctl -u filebeat -f
```
</details>

<details>
<summary><b>🔵 &nbsp;Prometheus target showing DOWN</b> &nbsp;<img src="https://img.shields.io/badge/Fix-UFW_port_9100-blue?style=flat-square"/></summary><br>

```bash
sudo ufw allow from [ELK_IP] to any port 9100
curl http://[SERVER_IP]:9100/metrics   # should return data
sudo systemctl status node_exporter
```
</details>

<details>
<summary><b>⚫ &nbsp;MySQL remote connection refused</b> &nbsp;<img src="https://img.shields.io/badge/Fix-bind--address+UFW-black?style=flat-square"/></summary><br>

```bash
# Check bind-address = 0.0.0.0 in mysqld.cnf
grep bind-address /etc/mysql/mysql.conf.d/mysqld.cnf
sudo ufw allow from [WEB_IP] to any port 3306
sudo systemctl restart mysql
```
</details>
